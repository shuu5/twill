3.2 実装 (pseudo-bash)

#!/bin/bash
# spawn-tmux.sh — 3 階層共通 spawn helper (~50 行)
# verified: tmux new-window -d -n -c flags tmux(1)

set -euo pipefail
source "$(dirname "$0")/mailbox.sh"

main {
  local window_name skill args cwd mailbox_from
  while [ $# -gt 0 ]; do
    case "$1" in
      --window-name) window_name="$2"; shift 2 ;;
      --skill) skill="$2"; shift 2 ;;
      --args) args="$2"; shift 2 ;;
      --cwd) cwd="$2"; shift 2 ;;
      --mailbox-from) mailbox_from="$2"; shift 2 ;;
      *) shift ;;
    esac
  done

  # 1. window 名 length check (50 文字上限、tmux 制約)
  [ "${#window_name}" -le 50 ] || _fail "window name too long: $window_name"

  # 2. 既存 window 確認 (Inv I-1 idempotent guard)
  if tmux list-windows -a -F '#{window_name}' | grep -qx "$window_name"; then
    _emit_admin "spawn-skip" "{\"window\":\"$window_name\",\"reason\":\"already exists\"}"
    return 0
  fi

  # 3. plugin 化 twl の場合、--plugin-dir で起動
  local plugin_dir="${TWILL_PLUGIN_DIR:-$CLAUDE_PROJECT_DIR/plugins/twl}"

  # 4. tmux new-window (detached、cwd 指定、env 変数注入)
  tmux new-window \
    -d \
    -n "$window_name" \
    -c "${cwd:-$CLAUDE_PROJECT_DIR}" \
    -e "TWL_PHASER_NAME=$window_name" \
    -- "claude --plugin-dir $plugin_dir"
  # ↑ tmux man verified: -d=detached, -n=name, -c=start-directory, -e=env variable
  # ↑ TWL_PHASER_NAME は atomic SKILL.md Step 3 が mailbox path 解決に使用
  #   (Phase I 2026-05-15 確定、atomic-verification.html §3.1 参照)

  # 5. slash command inject (Claude が起動して prompt ready になるまで wait)
  _wait_session_ready "$window_name" 30 || _fail "session not ready after 30s"

  # 6. inject /twl:<skill> <args>
  local cmd="/${skill} ${args:-}"
  tmux send-keys -t "$window_name" "$cmd" Enter
  # ↑ send-keys: Enter key を C-m として inject (公式 syntax)

  # 7. spawn-completed mail emit
  _emit_admin "spawn-completed" "{\"window\":\"$window_name\",\"skill\":\"$skill\"}"
}

_wait_session_ready {
  # Phase I 2026-05-15 確定: 3 条件 AND 判定 (CLAUDE.md feedback_inject_queue_verification 知見反映)
  # (a) pane_in_mode=0 (idle、edit mode でない)
  # (b) "Press up to edit queued" 不在 (旧 inject の queue 残留検知)
  # (c) idle 持続時間 > IDLE_MIN_SEC (default 2s、startup initialization 中 false ready 防止)
  local window="$1"; local timeout="$2"; local elapsed=0
  local idle_min_sec="${IDLE_MIN_SEC:-2}"
  local idle_accumulated=0

  while [ "$elapsed" -lt "$timeout" ]; do
    # (a) pane_in_mode=0 check
    local in_mode=$(tmux list-windows -a -F '#{window_name}|#{pane_in_mode}' \
                      | grep "^${window}|" | cut -d'|' -f2)
    if [ "$in_mode" != "0" ]; then
      idle_accumulated=0; sleep 1; elapsed=$((elapsed+1)); continue
    fi

    # (b) "Press up to edit queued" 不在 check (queue 残留検知)
    local pane_tail=$(tmux capture-pane -t "$window" -p -S -5 2>/dev/null | tail -3)
    if echo "$pane_tail" | grep -q "Press up to edit"; then
      # queue 残留 → Enter 送信して flush し再判定
      tmux send-keys -t "$window" Enter
      idle_accumulated=0; sleep 1; elapsed=$((elapsed+1)); continue
    fi

    # (c) idle 持続時間 check
    idle_accumulated=$((idle_accumulated+1))
    if [ "$idle_accumulated" -ge "$idle_min_sec" ]; then
      return 0  # 3 条件 AND 成立、ready + idle
    fi
    sleep 1; elapsed=$((elapsed+1))
  done
  return 1
}

_emit_admin {
  local event="$1"; local detail="$2"
  mailbox_emit "$mailbox_from" "administrator" "$event" "$detail"
}

_fail { echo "spawn-tmux.sh: $*" >&2; exit 1; }
main "$@"

§5. mailbox.sh helper 仕様 (~40 行)

API:
  • mailbox_emit <from> <to> <event> <detail_json> — atomic write 1 line
  • mailbox_read <mailbox-name> [--since <line-no>] — 未読 mail を stdout に出力 (read line-no を .read-offset file に保存)
  • mailbox_drain <mailbox-name> <callback-fn> — 未読 mail を 1 件ずつ callback に渡す
  • mailbox_archive <mailbox-name> — inbox.jsonl を archive/ に移動 (Wave 完了時)
#!/bin/bash
# mailbox.sh — file mailbox helper (~40 行)
# verified pattern: session-atomic-write.sh L41-50 (flock + mktemp + mv)
# 不変条件 T: file mailbox atomic write

set -euo pipefail
MAILBOX_ROOT="${TWILL_MAILBOX_ROOT:-.mailbox}"

mailbox_emit {
  local from="$1" to="$2" event="$3" detail="$4"
  local target_dir="$MAILBOX_ROOT/$to"
  mkdir -p "$target_dir"
  local lock_file="$target_dir/.lock"
  local inbox="$target_dir/inbox.jsonl"

  # JSON Lines 1 行 (jq -c で compact、改行なし)
  local line=$(jq -nc \
    --arg from "$from" --arg to "$to" \
    --arg event "$event" --argjson detail "$detail" \
    --arg ts "$(date -Iseconds)" \
    '{from: $from, to: $to, ts: $ts, event: $event, detail: $detail}')

  # flock atomic append (timeout 10s、fd 9)
  (
    flock -x -w 10 9 || exit 1
    echo "$line" >> "$inbox"
  ) 9>>"$lock_file"
}

mailbox_read {
  local mailbox="$1"
  local inbox="$MAILBOX_ROOT/$mailbox/inbox.jsonl"
  local offset_file="$MAILBOX_ROOT/$mailbox/.read-offset"
  local from=$(cat "$offset_file" 2>/dev/null || echo 0)

  [ -f "$inbox" ] || return 0

  # 共有 lock で read (write は exclusive で衝突防止)
  (
    flock -s -w 10 9 || exit 1
    tail -n +$((from + 1)) "$inbox"
    local total=$(wc -l < "$inbox")
    echo "$total" > "$offset_file"
  ) 9>>"$MAILBOX_ROOT/$mailbox/.lock"
}

6.1 event 種別 (本 spec で固定):

命名規約: mail event 名の phase-* / step-* prefix は canonical event 名として継続、role prefix (phaser-* / workflow-* / atomic-*) とは独立な名前空間。twl audit --vocabulary の forbidden 検出は role prefix 用、event 名 prefix は exclusion_context として除外する。
  • spawn 系: spawn-completed, spawn-failed, spawn-skip
  • phase 系: phase-started, phase-progress, phase-completed, phase-failed (canonical event 名、role prefix とは独立)
  • step 系 (atomic→phaser): step-started, step-completed, step-precheck-failed, step-postverify-failed (mail event 名の step-* prefix は canonical event 名として継続、role prefix とは独立)
  • gate 系: phase-gate-denied
  • tool 系: tool-completed (tool-sandbox-runner のみ administrator に送る、旧 tool-self-improve rename)
  • escalate 系: escalate-human-gate, budget-pause

§8. mail loss 防止 + archive 戦略

  • file 永続: disk 故障以外で loss しない (verified、JSON Lines + append-only)
  • backup: .mailbox/ は .gitignore 対象、git で持たない。重要 mail は archive で残す
  • archive: Wave 完了時に mailbox_archive で .mailbox/<name>/inbox.jsonl → .mailbox/archive/inbox-<name>-<ts>.jsonl
  • retention: 7 日 retention、cron で古い archive 削除
  • audit 統合: TWL_AUDIT=1 時に audit_snapshot(.mailbox, "mailbox-pre-archive") で .audit/<run-id>/ にもコピー

§10. subagent nesting 制約 (公式 verified)

公式制約 (verified): Claude Code subagent は 1 階層内側のみ spawn 可能、子 subagent の更なる spawn は禁止。
  • L2-3 (workflow / atomic / specialist) は内部で codex CLI 等の subagent を 1 階層内に持てる
  • L2-3 内 subagent が更に subagent spawn は 禁止
  • これが tmux で別 main session を spawn する根拠: 3 層 (administrator / phaser / L2-3) は全て別 main session、subagent 入れ子では実現不可
verified source: Claude Code 公式 docs (code.claude.com/docs/en/sub-agents.md L60/L348/L737)
