6b.2 STEP -1 実施手順

# Phase 1 PoC 開始時、user 明示承認後に実行
cd plugins/twl/hooks/
cp hooks.json hooks.json.bak                 # rollback 用 backup

# 必要最小限の hook のみ残す新 hooks.json を作成
cat > hooks.json <<'EOF'
{
  "hooks": {
    "PreToolUse": [
      { "matcher": "Edit|Write|NotebookEdit", "hooks": [{ "type": "command",
        "command": "${CLAUDE_PLUGIN_ROOT}/scripts/hooks/pre-tool-use-worktree-boundary.sh", "timeout": 10000 }] },
      { "matcher": "Bash", "hooks": [{ "type": "command",
        "command": "${CLAUDE_PLUGIN_ROOT}/scripts/hooks/pre-tool-use-host-safety.sh", "timeout": 10000 }] },
      { "matcher": "Bash", "if": "Bash(git commit *)", "hooks": [{ "type": "command",
        "command": "${CLAUDE_PLUGIN_ROOT}/scripts/hooks/pre-bash-commit-validate.sh", "timeout": 10000 }] },
      { "matcher": "Bash", "if": "Bash(git merge *)", "hooks": [{ "type": "command",
        "command": "${CLAUDE_PLUGIN_ROOT}/scripts/hooks/pre-bash-merge-guard.sh", "timeout": 10000 }] },
      { "matcher": "Edit|Write", "if": "Edit(deps.yaml)|Write(deps.yaml)", "hooks": [{ "type": "command",
        "command": "${CLAUDE_PLUGIN_ROOT}/scripts/hooks/pre-tool-use-deps-yaml-guard.sh", "timeout": 10000 }] }
    ],
    "PreCompact": [
      { "hooks": [{ "type": "command",
        "command": "${CLAUDE_PLUGIN_ROOT}/scripts/hooks/pre-compact-checkpoint.sh", "timeout": 30000 }] }
    ],
    "PostCompact": [
      { "hooks": [{ "type": "command",
        "command": "${CLAUDE_PLUGIN_ROOT}/scripts/hooks/post-compact-checkpoint.sh", "timeout": 30000 }] }
    ]
  }
}
EOF
# Claude Code 再起動 or /reload-plugins
# verify: 既存 hook の error 通知が消える、最小 hook のみ動作
