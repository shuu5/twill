# administrator main session гҒ® polling cycle (CronCreate 25min зҷәзҒ«гҒ§е®ҹиЎҢ)
admin_polling_cycle() {
  # --- READ phase ---
  # 1. иҮӘеҲҶгҒ® mailbox (pilot гҒӢгӮүгҒ®йӣҶзҙ„ report) гӮ’иӘӯгӮҖ
  ADMIN_MBOX="$REPO_ROOT/.mailbox/administrator/inbox.jsonl"
  PROCESSED_FILE="$REPO_ROOT/.mailbox/administrator/.processed"
  last_line=$(cat "$PROCESSED_FILE" 2>/dev/null || echo 0)
  new_mails=$(tail -n +$((last_line + 1)) "$ADMIN_MBOX")
  total_lines=$(wc -l < "$ADMIN_MBOX")
  echo "$total_lines" > "$PROCESSED_FILE"

  # 2. GitHub Project Board гҒ®е…Ё Issue Status гӮ’еҸ–еҫ—
  current_statuses=$(gh project item-list 6 --owner shuu5 --format json | \
    jq -c '.items[] | {num: .content.number, status: .status, title: .content.title}')

  # 3. гӮўгӮҜгғҶгӮЈгғ–гҒӘ tmux pilot / worker window гӮ’еҲ—жҢҷ
  active_windows=$(tmux list-windows -F '#{window_name}' | grep -E '^(phase-|worker-)')

  # --- JUDGE + WRITE phase ---
  # 4. spawn еҲӨж–ӯ: status йҒ·з§»е®ҢдәҶ + active дёҚеңЁ вҶ’ ж¬Ў phase spawn
  echo "$current_statuses" | while read -r item; do
    issue=$(echo "$item" | jq -r '.num')
    status=$(echo "$item" | jq -r '.status')

    # idempotent guard: ж—ўгҒ«еҗҢ Issue гҒ® pilot гҒҢеӢ•гҒ„гҒҰгҒ„гӮҢгҒ° skip
    if echo "$active_windows" | grep -q "phase-.*-${issue}$"; then
      continue
    fi

    # ж¬Ў phase еҲӨе®ҡ
    case "$status" in
      Idea)         next_phase="phase-explore" ;;
      Explored)     next_phase="phase-refine"  ;;
      Refined)      next_phase="phase-impl"    ;;
      Implementing) next_phase="phase-pr"      ;;
      "PR Reviewed"|Merged) continue ;;  # actions гҒҢеҮҰзҗҶ
    esac
    [ -z "$next_phase" ] && continue

    # дёҰеҲ—еәҰдёҠйҷҗ check (дҫӢ: еҗҢжҷӮ active pilot гҒҜ 3 гҒҫгҒ§)
    active_pilot_count=$(echo "$active_windows" | grep -cE '^phase-')
    if [ "$active_pilot_count" -ge 3 ]; then
      echo "[ADMIN] parallelism limit reached, defer issue=$issue" >&2
      continue
    fi

    # tmux spawn
    tmux new-window -n "${next_phase}-${issue}" -- \
      bash -c "cd $REPO_ROOT && claude code --skill ${next_phase} --arg issue_num=${issue}"
    echo "[ADMIN] spawned ${next_phase}-${issue}" >&2
  done

  # 5. failure / escalate еҮҰзҗҶ (mail еҶ…гҒ« failure flag гҒҢгҒӮгӮҢгҒ° user notify)
  echo "$new_mails" | jq -c 'select(.event == "failure" or .escalate == true)' | while read -r mail; do
    # AskUserQuestion or Discord йҖҡзҹҘ (memory гҒ«гҒӮгӮӢ reference_discord_dm.md еҸӮз…§)
    issue=$(echo "$mail" | jq -r '.issue')
    reason=$(echo "$mail" | jq -r '.reason')
    echo "[ADMIN] ESCALATE: issue=$issue reason=$reason" >&2
    # вҶ’ гҒ“гҒ“гҒ§ user notify (жң¬ architecture гҒ§гҒҜ AskUserQuestion or Discord DM)
  done

  # 6. cleanup (е®ҢдәҶжёҲ pilot window гӮ’ close + mailbox archive)
  echo "$active_windows" | while read -r win; do
    # status гҒҢзөӮз«Ҝ (PR Reviewed or Merged) гҒ® Issue гҒ® pilot гҒҜ close
    issue=$(echo "$win" | grep -oE '[0-9]+$')
    [ -z "$issue" ] && continue
    status_of_issue=$(echo "$current_statuses" | jq -r --arg n "$issue" '. | select(.num == ($n | tonumber)) | .status')
    case "$status_of_issue" in
      "PR Reviewed"|Merged) tmux kill-window -t "$win" ;;
    esac
  done

  # еҸӨгҒ„ mailbox гӮ’ archive (7 ж—ҘзөҢйҒҺгҒ—гҒҹгӮӮгҒ®гӮ’ .mailbox/archive/ гҒ«з§»еӢ•)
  find "$REPO_ROOT/.mailbox" -maxdepth 2 -name "inbox.jsonl" -mtime +7 \
    -exec mv {} "$REPO_ROOT/.mailbox/archive/" \;
}

}

# еҗ„ step гҒ® verification rule (case гҒ§ dispatch)
step_run() {
  local step_name="$1"; shift
  local args=("$@")

  # parse key=value pairs
  declare -A opts
  for arg in "${args[@]}"; do opts["${arg%%=*}"]="${arg#*=}"; done

  echo "[step.sh] ===== step=$step_name BEGIN =====" >&2

  # 1. pre-check (state snapshot)
  local pre_snapshot=$(_step_pre_check "$step_name" opts) || _step_fail "$step_name" pre "$pre_snapshot"
  echo "[step.sh] pre: $pre_snapshot" >&2

  # 2. exec (LLM гҒҜе‘јгҒіеҮәгҒ—е…ғгҒ® skill еҶ…гҒ§е®ҹиЎҢжёҲгҒҝгҖҒstep.sh гҒҜ LLM дҪңжҘӯгҒ® verify гҒ®гҒҝ)
  # exec иҮӘдҪ“гҒҜ LLM гҒҢ skill еҶ…гҒ§иЎҢгҒҶ = step.sh гҒ®иІ¬еӢҷеӨ–
  # (step_run гҒҜ post-verify гҒЁ report гҒ®гҒҝгӮ’и«ӢгҒ‘иІ гҒҶ)

  # 3. post-verify (step еӣәжңүгҒ® rule)
  local verify_result=$(_step_post_verify "$step_name" "$pre_snapshot" opts) || \
    _step_fail "$step_name" post "$verify_result"
  echo "[step.sh] post: $verify_result" >&2

  # 4. report (pilot mailbox гҒ«жҲҗеҠҹ mail гӮ’жӣёгҒҸ)
  _step_report "$step_name" success "$verify_result" opts
  echo "[step.sh] ===== step=$step_name OK =====" >&2
}

# step еӣәжңү rule гҒ® dispatch
_step_post_verify() {
  local step="$1"; local pre="$2"; local -n opts_ref="$3"
  case "$step" in
    test-scaffold)
      _verify_red_test_added "$pre" "${opts_ref[file]}"
      ;;
    green-impl)
      _verify_src_changed_and_green "$pre" "${opts_ref[file]}"
      ;;
    check)
      _verify_lint_typecheck "${opts_ref[file]}"
      ;;
    pr-test)
      _verify_pr_test_pass "${opts_ref[issue]}"
      ;;
    ac-verify)
      _verify_ac_checkbox_flip "${opts_ref[issue]}"
      ;;
    phase-review)
      _verify_agent_return_value "${opts_ref[issue]}"
      ;;
    *)
      echo "ERROR: unknown step=$step" >&2
      return 1
      ;;
  esac
}

# еҖӢеҲҘ verification й–ўж•° (rule per step)
_verify_red_test_added() {
  local pre="$1"; local file="$2"
  local pre_count=$(echo "$pre" | jq -r '.test_count')
  local post_count=$(pytest --collect-only -q 2>&1 | tail -1 | awk '{print $1}')
  [ "$post_count" -gt "$pre_count" ] || { echo "no new test"; return 1; }
  pytest -x --tb=no -q 2>&1 | tail -1 | grep -qv "passed" || { echo "test not RED"; return 1; }
  echo "{\"red_count\": $((post_count - pre_count))}"
}

_verify_src_changed_and_green() {
  local pre="$1"; local file="$2"
  [ "$(git diff --stat src/ | wc -l)" -gt 0 ] || { echo "no src/ change"; return 1; }
  pytest -x --tb=short 2>&1 >/dev/null || { echo "test still RED"; return 1; }
  echo "{\"green\": true}"
}

# pilot mailbox гҒ«жӣёгҒҚиҫјгҒҝ
_step_report() {
  local step="$1"; local status="$2"; local detail="$3"
  local pilot_mbox="$REPO_ROOT/.mailbox/${PILOT_NAME:-pilot-unknown}/inbox.jsonl"
  (
    flock -x 200
    jq -nc \
      --arg from "${WORKER_NAME:-worker-unknown}" \
      --arg ts "$(date -u +%FT%TZ)" \
      --arg step "$step" \
      --arg status "$status" \
      --argjson detail "$detail" \
      '{from:$from, ts:$ts, step:$step, status:$status, detail:$detail}' \
      >> "$pilot_mbox"
  ) 200>"$pilot_mbox.lock"
}

# fail жҷӮгҒ®еҮҰзҗҶ
_step_fail() {
  local step="$1"; local phase="$2"; local reason="$3"
  _step_report "$step" "failure" "{\"phase\":\"$phase\",\"reason\":\"$reason\"}"
  echo "[step.sh] FAIL step=$step phase=$phase reason=$reason" >&2
  exit 1
}
