3.2 各 EXP file の構造規約

#!/bin/bash
# EXP-001: PreToolUse hook 'matcher' (singular) verified by official docs
#
# verify target: spec claim "matcher 単数が正しい (旧 'matchers' 複数形は誤)"
# verify source (verified): https://code.claude.com/docs/en/hooks
# expected: hook config with `matcher: "Bash"` is accepted, with `matchers: ["Bash"]` is rejected
# related spec: gate-hook.html §1.1, research-findings.html §7

set -euo pipefail

EXP_ID="EXP-001"
RUN_ID="${TWL_AUDIT_DIR:-/tmp}/experiments"
mkdir -p "$RUN_ID"
RESULT_FILE="$RUN_ID/${EXP_ID}.json"

# === step 1: setup ===
TMPDIR=$(mktemp -d)
trap "rm -rf $TMPDIR" EXIT

# === step 2: exec (with both schemas) ===
#

# === step 3: assert ===
PASS=true
REASON=""

# === step 4: report ===
jq -n \
  --arg id "$EXP_ID" \
  --arg ts "$(date -Iseconds)" \
  --arg src "https://code.claude.com/docs/en/hooks" \
  --argjson pass "$PASS" \
  --arg reason "$REASON" \
  '{exp_id: $id, ts: $ts, verify_source: $src, pass: $pass, reason: $reason}' \
  > "$RESULT_FILE"

if [[ "$PASS" == "true" ]]; then
  echo "[$EXP_ID] PASS"
  exit 0
else
  echo "[$EXP_ID] FAIL: $REASON" >&2
  exit 1
fi

3.3 audit 統合

experiments/run-all.sh:

#!/bin/bash
set -euo pipefail

export TWL_AUDIT=1
export TWL_AUDIT_DIR="$(pwd)/.audit/$(date +%Y%m%d-%H%M%S)-experiments"
mkdir -p "$TWL_AUDIT_DIR/experiments"

# 1. bats 化可能 (test-fixtures 内)
bats test-fixtures/experiments/hooks/

# 2. smoke test (top-level)
for exp in experiments/EXP-*.smoke.sh; do
  bash "$exp" || echo "[FAIL] $exp"
done

# 3. index 集約
jq -s '{run_id: env.TWL_AUDIT_DIR, experiments: ., generated_at: now | todate}' \
  "$TWL_AUDIT_DIR/experiments/"*.json > "$TWL_AUDIT_DIR/experiments/index.json"

echo "All experiments completed. Results: $TWL_AUDIT_DIR/experiments/"
